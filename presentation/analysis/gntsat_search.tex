\frametitle{How GNTSAT searches}
\begin{enumerate}
	\item Pick some random points
	\item \label{enm:select} Select the points (lower points have better chance)
	\item Crossover
	\item Noisy greedy descent on children
	\item Repeat from \ref{enm:select}
\end{enumerate}
