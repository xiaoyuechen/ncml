\begin{frame}
\frametitle{The search space}

\begin{itemize}
	\item $n$: number of Boolean variables
	\item $m$: number of clauses
	\item $h$: number of unsatisfied clauses. $h \in [0, m]$
	\item $x$: a bit string of length $n$ (e.g.
	      $0111011$). $2^n$ possible
	      $x$'s
\end{itemize}
